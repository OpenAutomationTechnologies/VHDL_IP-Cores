# Memory mapping:
# port A : 0x0000 - 0x03FF
# port B : 0x0400 - 0x07FF

# Start with simple memory test
WR d 00000000 11111111
WR d 00000001 22222222
WR d 00000002 33333333
WR d 00000003 44444444
WR d 00000004 55555555
WR d 00000005 66666666
WR d 00000006 77777777
WR d 00000007 88888888
NOP
ASS d 00000000 11111111
ASS d 00000001 22222222
ASS d 00000002 33333333
ASS d 00000003 44444444
ASS d 00000004 55555555
ASS d 00000005 66666666
ASS d 00000006 77777777
ASS d 00000007 88888888

# Write something to port A ...
WR d 00000000 DEADBEEF
WR d 0000000F DEADC0DE
WR d 000000FF 0C0FFEE0
WR d 000003FF 12345678
NOP
# And read it from port B
ASS d 00000400 DEADBEEF
ASS d 0000040F DEADC0DE
ASS d 000004FF 0C0FFEE0
ASS d 000007FF 12345678

# Write something to port B ...
WR d 00000400 12345678
WR d 0000040F 0C0FFEE0
WR d 000004FF DEADC0DE
WR d 000007FF DEADBEEF
NOP
# And read it from port A
ASS d 00000000 12345678
ASS d 0000000F 0C0FFEE0
ASS d 000000FF DEADC0DE
ASS d 000003FF DEADBEEF

FIN
